def MyLasso(X, y, lam_seq, maxit = 500):
    
    # X: n-by-p design matrix without the intercept 
    # y: n-by-1 response vector 
    # lam_seq: sequence of lambda values 
    # maxit: number of updates for each lambda 
 
    # Center/Scale X
    # Center y
  
    n, p = X.shape
    nlam = len(lam_seq)

    # Record the corresponding means and scales
    y_mean=np.mean(y)
    X_mean=np.mean(X,axis=0)
    X_std=np.std(X,axis=0)
    Xs=X-X_mean
    Xs=Xs/X_std

    y=y-y_mean
         
    # Initilize coef vector b and residual vector r
    b = np.zeros(p)
    r = y
    r=r.reshape(-1,1)
    B = np.zeros((nlam, p + 1))
    # Triple nested loop
    for m in range(nlam):
        lam = 2 * n * lam_seq[m]
        for step in range(maxit):
            for j in range(p):
                Xs_j = Xs[:, j].reshape(-1,1)
                r = r + Xs_j * b[j]
                b[j] = one_var_lasso(r, Xs_j, lam)
                r = r - Xs_j * b[j]
        B[m, 1:] = b 
    
    ##############################
    # Scale back the coefficients;
    ##############################
    B[:,1:14]=B[:,1:14]/X_std
    
    ##############################
    # calculate b0;
    ##############################
    b_0=y_mean-(np.dot(B[:,1:14],X_mean))
    B[:,0]=b_0

    return(B.T)

def one_var_lasso(r, x, lam):
    # the code is written with r, x being n-by-1 vectors
    xx = x.T @ x
    xr = x.T @ r
    b = (abs(xr) - lam/2)/xx
    if b < 0:
        b = 0
    b = np.sign(xr) * b
    return(b)
